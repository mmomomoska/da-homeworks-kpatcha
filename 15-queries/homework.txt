SELECT orderNumber, productCode, (quantityordered * priceEach) AS maxprice
FROM orderdetails
ORDER BY maxprice DESC
LIMIT 10

Задача 2
SELECT orderNumber, SUM(quantityordered * priceEach) AS total
FROM orderdetails
GROUP BY orderNumber
HAVING total > 59000

Задача 3

SELECT orderdetails.orderNumber, orderDate, status, SUM(quantityordered * priceEach) AS total
FROM orders
INNER JOIN orderdetails
ON orders.orderNumber = orderdetails.orderNumber
GROUP BY orderNumber
HAVING total > 59000

Задача 4

SELECT contactFirstName, contactLastName, country, orderdetails.orderNumber, orderDate, status, SUM(quantityordered * priceEach) AS total
FROM orders
INNER JOIN orderdetails
ON orders.orderNumber = orderdetails.orderNumber
INNER JOIN customers
ON customers.customerNumber = orders.customerNumber
GROUP BY orderNumber
HAVING total > 59000

Задача 5

SELECT productName, SUM(quantityordered * priceEach) AS pricetotal
FROM 'orderdetails'
INNER JOIN 'products'
ON products.productCode = oderdetails.productCode
GROUP BY productName
ORDER BY pricetotal DESC
LIMIT 10
